###mpp:
 sudo dpkg -i librockchip-mpp1_1.5.0-1_arm64.deb librockchip-mpp-dev_1.5.0-1_arm64.deb librockchip-vpu0_1.5.0-1_arm64.deb
 sudo dpkg -i librockchip-mpp1-dbgsym_1.5.0-1_arm64.deb rockchip-mpp-demos_1.5.0-1_arm64.deb rockchip-mpp-demos-dbgsym_1.5.0-1_arm64.deb
 ls -la /dev/mpp_service
 ls /sys/devices/platform/ | grep rknpu
 注意：系统没有/dev/rknpu*，而是有/dev/mpp_service
 这是 RK3566/RK3568/RK3588 等较新芯片的标准架构：
 - 不再暴露单独的 /dev/rknpu0, /dev/rknpu1 等设备
 - 统一通过 mpp_service 进行多媒体和 NPU 任务调度
 - 驱动层在内核中处理多核 NPU 的负载均衡

###ffmpeg-rk 支持rkmpp的ffmpeg:
 git clone --depth=1 https://github.com/nyanmisaka/ffmpeg-rockchip.git ffmpeg-rockchip
 chmod +x configure
 ls /usr/lib/aarch64-linux-gnu/librockchip_mpp.so
 ls /usr/include/rockchip/rk_mpi.h
 ./configure \
  --prefix=/opt/ffmpeg-rk \
  --enable-gpl \
  --enable-version3 \
  --enable-nonfree \
  --enable-libdrm \
  --enable-rkmpp \
  --enable-rkrga \
  --enable-alsa \
  --extra-cflags="-I/usr/include/rockchip" \
  --extra-ldflags="-L/usr/lib/aarch64-linux-gnu"

 grep -E "rkmpp|rkrga|libdrm" ffbuild/config.log 确认：
 CONFIG_RKMPP=yes
 CONFIG_RKRGA=yes
 CONFIG_LIBDRM=yes

 make -j$(nproc)
 sudo make install
 ls /opt/ffmpeg-rk/bin
 /opt/ffmpeg-rk/bin/ffmpeg -decoders | grep rkmpp
